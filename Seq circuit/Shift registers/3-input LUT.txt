module top_module (
    input clk,
    input enable,
    input S,
    input A, B, C,
    output Z );
        
    reg [7:0] Q; // 8-bit shift register
    always @(posedge clk) begin
        if (enable)
            Q <= {Q[6:0], S};   // Shift left, MSB shifted in first
    end

    // 8:1 mux selects one bit of Q
    assign Z = Q[{A, B, C}];

endmodule
